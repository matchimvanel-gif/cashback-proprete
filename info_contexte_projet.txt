```js
// Cron '0 0 1 * *' (1er jour 00h)
exports.updateMontantContrat = functions.pubsub.schedule('0 0 1 * *').onRun(async () => {
  const etablissements = await db.collection('etablissements').get();
  etablissements.forEach(doc => {
    const data = doc.data();
    if (data.dateRenouvellement && (new Date() - data.dateRenouvellement.toDate()) > 30*24*60*60*1000) {
      // Calcul basé cote + volume depots derniers 30j from @infoProjet
      const nouveauMontant = calculerMontant(data.cote_moyenne, volumeDepots);
      doc.ref.update({ montantContrat: nouveauMontant, dateRenouvellement: admin.firestore.Timestamp.now() });
    }
  });
});
```
