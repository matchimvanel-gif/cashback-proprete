**FIRESTORE SECURITY RULES - CASHBACK PROPRETÉ** (copier dans firestore.rules)

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // utilisateurs: chacun son doc
    match /utilisateurs/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow read: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // depots: citoyen lit siens, responsable crée, hyzakam lit
    match /depots/{depotId} {
      allow read: if request.auth != null && resource.data.id_citoyen == request.auth.uid;
      allow create: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'responsable';
      allow read: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // coupons: citoyen lit siens, établissement valide, hyzakam lit
    match /coupons/{couponId} {
      allow read: if request.auth != null && resource.data.id_citoyen == request.auth.uid;
      allow update: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'etablissement' && resource.data.statut == 'disponible';
      allow read: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // etablissements: lecture publique offres, hyzakam write
    match /etablissements/{etabId} {
      allow read: if true;
      allow write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // configuration: read authenticated
    match /configuration/{configId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // bacs: responsable son bac, hyzakam RW
    match /bacs/{bacId} {
      allow read, update: if request.auth != null && request.auth.uid == bacId;
      allow read, write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // signalements: create all, hyzakam traite
    match /signalements/{signalId} {
      allow read, create: if request.auth != null;
      allow update: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // cote: read all
    match /cote/{coteId} {
      allow read: if true;
      allow write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }

    // partenaires/RespoID: read all, hyzakam RW
    match /partenaires/{id} { allow read: if true; allow write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam'; }
    match /RespoID/{id} { allow read: if true; allow write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam'; }

    // alertes: hyzakam only
    match /alertes/{alerteId} {
      allow read, write: if request.auth != null && get(/databases/$(database)/documents/utilisateurs/$(request.auth.uid)).data.role == 'hyzakam';
    }
  }
}
```

**Deploy**: `firebase deploy --only firestore:rules`
